# .github/workflows/main.yaml
name: Smart DevSecOps CI/CD Pipeline

on:
  push:
    branches: [ "main" ]
  workflow_dispatch:

jobs:
  security-pipeline:
    name: Full Security Pipeline
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write
      actions: write
      security-events: write

    steps:
      # ---------------------------
      # 1?? Cloner le dépôt
      # ---------------------------
      - name: Checkout code
        uses: actions/checkout@v4

      # ---------------------------
      # 2?? Analyse IaC (Terraform, K8s, etc.)
      # ---------------------------
      - name: 'SECURITY: Scan IaC files with Checkov'
        uses: bridgecrewio/checkov-action@master
        with:
          directory: ./kubernetes
          soft_fail: true

      # ---------------------------
      # 3?? Analyse SCA (Dépendances) avec Trivy
      # ---------------------------
      - name: 'SECURITY: Scan dependencies with Trivy (SCA)'
        uses: aquasecurity/trivy-action@master
        with:
          scan-type: 'fs'
          scan-ref: './app-bancaire'
          format: 'table'
          exit-code: '0'  # Continue même si vulnérabilités trouvées

      # ---------------------------
      # 4?? Construction de l'image Docker
      # ---------------------------
      - name: Build Docker image locally
        uses: docker/build-push-action@v5
        with:
          context: ./app-bancaire
          load: true
          tags: banking-api-local:${{ github.sha }}

      # ---------------------------
      # 5?? Scan de l'image Docker (Trivy Image)
      # ---------------------------
      - name: 'SECURITY: Scan Docker image with Trivy'
        uses: aquasecurity/trivy-action@master
        with:
          scan-type: 'image'
          image-ref: 'banking-api-local:${{ github.sha }}'
          format: 'table'
          exit-code: '0'

      # ---------------------------
      # 6?? Lancer l’application pour DAST (OWASP ZAP)
      # ---------------------------
      - name: Run application in background for DAST scan
        run: |
          docker run -d --name banking-app -p 5000:5000 banking-api-local:${{ github.sha }}
          echo "Waiting 20 seconds for the application to start..."
          sleep 20

      - name: 'SECURITY: Run OWASP ZAP Baseline Scan'
        uses: zaproxy/action-baseline@v0.10.0
        with:
          target: 'http://localhost:5000/'
          fail_action: false
          artifact_name: zap-scan-report
        continue-on-error: true

      # ---------------------------
      # 7?? Sauvegarder le rapport ZAP comme artefact
      # ---------------------------
      - name: Upload ZAP Scan Report
        uses: actions/upload-artifact@v4
        with:
          name: zap-scan-report
          path: ./zap-scan-report.html
        continue-on-error: true

      # ---------------------------
      # 8?? Connexion à GitHub Container Registry (GHCR)
      # ---------------------------
      - name: Log in to GitHub Container Registry
        if: success()
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.PAT }}

      # ---------------------------
      # 9?? Push de l'image validée dans GHCR
      # ---------------------------
      - name: Push validated image to GHCR
        if: success()
        run: |
          IMAGE_NAME=ghcr.io/${{ github.repository_owner }}/banking-api:${{ github.sha }}
          docker tag banking-api-local:${{ github.sha }} $IMAGE_NAME
          docker push $IMAGE_NAME

# Fichier Unique : policy-disallow-latest-tag.yaml

# --- PARTIE 1 : Le Modèle (La "Recette") ---
apiVersion: templates.gatekeeper.sh/v1
kind: ConstraintTemplate
metadata:
  name: k8sdisallowlatesttag
spec:
  crd:
    spec:
      names:
        kind: K8sDisallowLatestTag
  targets:
    - target: admission.k8s.gatekeeper.sh
      rego: |
        package k8sdisallowlatesttag

        violation[{"msg": msg}] {
          container := input.review.object.spec.containers[_]
          endswith(container.image, ":latest")
          msg := sprintf("Image <%v> uses the 'latest' tag in container <%v>, which is not allowed", [container.image, container.name])
        }

        violation[{"msg": msg}] {
          container := input.review.object.spec.initContainers[_]
          endswith(container.image, ":latest")
          msg := sprintf("Image <%v> uses the 'latest' tag in initContainer <%v>, which is not allowed", [container.image, container.name])
        }

        violation[{"msg": msg}] {
          container := input.review.object.spec.ephemeralContainers[_]
          endswith(container.image, ":latest")
          msg := sprintf("Image <%v> uses the 'latest' tag in ephemeralContainer <%v>, which is not allowed", [container.image, container.name])
        }

--- # <--- SÉPARATEUR TRÈS IMPORTANT
